/** * NO~BULL BOOKS — JWT VERIFICATION * ───────────────────────────────────────────────────────────────────────────── * Verifies Google Identity Services ID tokens (JWTs) issued by the Sign-In * widget on app.html. This is the authentication boundary for cross-domain * calls from nobull.consulting into the Apps Script web app. * * Why this exists: * Apps Script's Session.getActiveUser() does not return an email for * anonymous web app deployments called from another origin via JSONP. * The frontend instead sends the Google-issued JWT, and we verify it * here against Google's tokeninfo endpoint before trusting any claims. * * Required Script Properties: * GOOGLE_OAUTH_CLIENT_ID — the OAuth client ID used by app.html's * Google Identity Services widget. * * Required OAuth scope (already in appsscript.json): * https://www.googleapis.com/auth/script.external_request * ───────────────────────────────────────────────────────────────────────────── */ var GOOGLE_TOKENINFO_URL = 'https://oauth2.googleapis.com/tokeninfo?id_token='; var TOKEN_CACHE_SECONDS = 300; // 5 minutes — must be < token lifetime var GOOGLE_VALID_ISSUERS = ['https://accounts.google.com', 'accounts.google.com']; /** * verifyGoogleJwt(token) * Returns the verified payload or throws on any verification failure. * * Performs: * - Sanity check on token shape * - Cache lookup (avoids re-validating the same token within 5 minutes) * - Round-trip to Google's tokeninfo endpoint, which validates signature, * issuer, expiry, and audience server-side * - Local audience check against GOOGLE_OAUTH_CLIENT_ID (defence in depth) * - email_verified check * * The returned payload is a plain object with at least: * { sub, email, email_verified, name, picture, aud, iss, exp, iat } */ function verifyGoogleJwt(token) { if (!token || typeof token !== 'string') { throw new Error('JWT_MISSING'); } // A valid JWT has three base64url segments separated by dots. var parts = token.split('.'); if (parts.length !== 3) throw new Error('JWT_MALFORMED'); var expectedAud = PropertiesService.getScriptProperties() .getProperty('GOOGLE_OAUTH_CLIENT_ID'); if (!expectedAud) throw new Error('JWT_NO_CLIENT_ID_CONFIGURED'); // Cache by the JWT signature segment (last part) — every distinct token // has a distinct signature, and the segment is short enough for a cache key. var cacheKey = 'jwt_' + parts[2].substring(0, 200); var cache = CacheService.getScriptCache(); var cached = cache.get(cacheKey); if (cached) { var cachedPayload = JSON.parse(cached); if (cachedPayload && cachedPayload.exp * 1000 > Date.now()) { return cachedPayload; } } var resp = UrlFetchApp.fetch(GOOGLE_TOKENINFO_URL + encodeURIComponent(token), { muteHttpExceptions: true }); var code = resp.getResponseCode(); if (code !== 200) { // Google returns 400 with { error_description: "..." } for invalid tokens var detail = ''; try { var errBody = JSON.parse(resp.getContentText()); detail = errBody.error_description || errBody.error || ''; } catch(_) { /* ignore */ } throw new Error('JWT_REJECTED_BY_GOOGLE' + (detail ? ': ' + detail : '')); } var payload; try { payload = JSON.parse(resp.getContentText()); } catch(e) { throw new Error('JWT_BAD_RESPONSE'); } // Defence in depth — Google should already check these, but verify locally. if (GOOGLE_VALID_ISSUERS.indexOf(payload.iss) === -1) { throw new Error('JWT_BAD_ISSUER: ' + payload.iss); } if (payload.aud !== expectedAud) { throw new Error('JWT_BAD_AUDIENCE'); } // tokeninfo returns exp as a string of seconds; coerce. var expSec = parseInt(payload.exp, 10); if (!expSec || expSec * 1000 < Date.now()) { throw new Error('JWT_EXPIRED'); } payload.exp = expSec; if (!payload.email) { throw new Error('JWT_NO_EMAIL'); } // tokeninfo returns email_verified as the string "true"/"false". if (payload.email_verified === 'false' || payload.email_verified === false) { throw new Error('JWT_EMAIL_NOT_VERIFIED'); } // Cache for short window so repeat calls within a session are fast. // Cap at the actual remaining lifetime so we never serve an expired token. var ttl = Math.min(TOKEN_CACHE_SECONDS, expSec - Math.floor(Date.now() / 1000)); if (ttl > 0) { cache.put(cacheKey, JSON.stringify(payload), ttl); } return payload; } /** * verifyJwtFromRequest(e) * Convenience for use from doGet/doPost. Pulls the credential from the * `credential` parameter, verifies it, and returns the lowercased email. * Throws if absent or invalid. */ function verifyJwtFromRequest(e) { var credential = e && e.parameter && e.parameter.credential; if (!credential && e && e.postData && e.postData.contents) { try { var body = JSON.parse(e.postData.contents); credential = body.credential; } catch(_) { /* ignore */ } } var payload = verifyGoogleJwt(credential); return String(payload.email).toLowerCase().trim(); } // ───────────────────────────────────────────────────────────────────────────── // Editor helpers // ───────────────────────────────────────────────────────────────────────────── /** Run from the editor to set the OAuth client ID. */ function setGoogleOAuthClientId(clientId) { PropertiesService.getScriptProperties().setProperty('GOOGLE_OAUTH_CLIENT_ID', clientId); Logger.log('GOOGLE_OAUTH_CLIENT_ID set.'); } /** Run from the editor with a real JWT pasted from the browser to test. */ function debug_verifyJwt(token) { try { var p = verifyGoogleJwt(token); Logger.log('OK — email: ' + p.email + ', sub: ' + p.sub); return p; } catch(e) { Logger.log('FAIL: ' + e); return { error: e.toString() }; } }
